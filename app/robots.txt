import { MetadataRoute } from 'next';

export default function robots() {
  const baseUrl = process.env.NEXT_PUBLIC_BASE_URL || 'https://metalcre.vercel.app/';
  
  return {
    rules: {
      userAgent: '*',
      allow: '/',
      disallow: ['/private/', '/admin/'],
    },
    sitemap: `${baseUrl}/sitemap.xml`,
  };
}
